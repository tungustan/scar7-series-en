import pygame import time def play_music(file_path): # 믹서 초기화 pygame.mixer.init() # 음악 로드 pygame.mixer.music.load(file_path) # 재생 시작 print(f"재생 중: {file_path}") pygame.mixer.music.play() # 음악이 재생되는 동안 프로그램이 종료되지 않게 대기 while pygame.mixer.music.get_busy(): time.sleep(1) # 실행 (본인의 파일명으로 수정하세요) if __name__ == "__main__": play_music("your_audio_file.mp3")
